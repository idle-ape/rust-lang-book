11//! The implementations of the `Standard` distribution for integer types.
12
13use {Rng};
14use distributions::{Distribution, Standard};
15
16impl Distribution<u8> for Standard {
17    #[inline]
18    fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> u8 {
19        rng.next_u32() as u8
20    }
21}
22
23impl Distribution<u16> for Standard {
24    #[inline]
25    fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> u16 {
26        rng.next_u32() as u16
27    }
28}
29
30impl Distribution<u32> for Standard {
31    #[inline]
32    fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> u32 {
33        rng.next_u32()
34    }
35}
36
37impl Distribution<u64> for Standard {
38    #[inline]
39    fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> u64 {
40        rng.next_u64()
41    }
42}
43
44#[cfg(feature = "i128_support")]
45impl Distribution<u128> for Standard {
46    #[inline]
47    fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> u128 {
48        // Use LE; we explicitly generate one value before the next.
49        let x = rng.next_u64() as u128;
50        let y = rng.next_u64() as u128;
51        (y << 64) | x
52    }
53}
54
55impl Distribution<usize> for Standard {
56    #[inline]
57    #[cfg(any(target_pointer_width = "32", target_pointer_width = "16"))]
58    fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> usize {
59        rng.next_u32() as usize
60    }
61
62    #[inline]
63    #[cfg(target_pointer_width = "64")]
64    fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> usize {
65        rng.next_u64() as usize
66    }
67}
68
69macro_rules! impl_int_from_uint {
70    ($ty:ty, $uty:ty) => {
71        impl Distribution<$ty> for Standard {
72            #[inline]
73            fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> $ty {
74                rng.gen::<$uty>() as $ty
75            }
76        }
77    }
78}
79
80impl_int_from_uint! { i8, u8 }
81impl_int_from_uint! { i16, u16 }
82impl_int_from_uint! { i32, u32 }
83impl_int_from_uint! { i64, u64 }
84#[cfg(feature = "i128_support")] impl_int_from_uint! { i128, u128 }
85impl_int_from_uint! { isize, usize }
