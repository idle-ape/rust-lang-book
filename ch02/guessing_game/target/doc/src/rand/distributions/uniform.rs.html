11//! A distribution uniformly sampling numbers within a given range.
12//!
13//! [`Uniform`] is the standard distribution to sample uniformly from a range;
14//! e.g. `Uniform::new_inclusive(1, 6)` can sample integers from 1 to 6, like a
15//! standard die. [`Rng::gen_range`] supports any type supported by
16//! [`Uniform`].
17//!
18//! This distribution is provided with support for several primitive types
19//! (all integer and floating-point types) as well as `std::time::Duration`,
20//! and supports extension to user-defined types via a type-specific *back-end*
21//! implementation.
22//!
23//! The types [`UniformInt`], [`UniformFloat`] and [`UniformDuration`] are the
24//! back-ends supporting sampling from primitive integer and floating-point
25//! ranges as well as from `std::time::Duration`; these types do not normally
26//! need to be used directly (unless implementing a derived back-end).
27//!
28//! # Example usage
29//!
30//! ```
31//! use rand::{Rng, thread_rng};
32//! use rand::distributions::Uniform;
33//! 
34//! let mut rng = thread_rng();
35//! let side = Uniform::new(-10.0, 10.0);
36//! 
37//! // sample between 1 and 10 points
38//! for _ in 0..rng.gen_range(1, 11) {
39//!     // sample a point from the square with sides -10 - 10 in two dimensions
40//!     let (x, y) = (rng.sample(side), rng.sample(side));
41//!     println!("Point: {}, {}", x, y);
42//! }
43//! ```
44//!
45//! # Extending `Uniform` to support a custom type
46//!
47//! To extend [`Uniform`] to support your own types, write a back-end which
48//! implements the [`UniformSampler`] trait, then implement the [`SampleUniform`]
49//! helper trait to "register" your back-end. See the `MyF32` example below.
50//!
51//! At a minimum, the back-end needs to store any parameters needed for sampling
52//! (e.g. the target range) and implement `new`, `new_inclusive` and `sample`.
53//! Those methods should include an assert to check the range is valid (i.e.
54//! `low < high`). The example below merely wraps another back-end.
55//!
56//! ```
57//! use rand::prelude::*;
58//! use rand::distributions::uniform::{Uniform, SampleUniform,
59//!         UniformSampler, UniformFloat};
60//!
61//! struct MyF32(f32);
62//!
63//! #[derive(Clone, Copy, Debug)]
64//! struct UniformMyF32 {
65//!     inner: UniformFloat<f32>,
66//! }
67//!
68//! impl UniformSampler for UniformMyF32 {
69//!     type X = MyF32;
70//!     fn new(low: Self::X, high: Self::X) -> Self {
71//!         UniformMyF32 {
72//!             inner: UniformFloat::<f32>::new(low.0, high.0),
73//!         }
74//!     }
75//!     fn new_inclusive(low: Self::X, high: Self::X) -> Self {
76//!         UniformSampler::new(low, high)
77//!     }
78//!     fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> Self::X {
79//!         MyF32(self.inner.sample(rng))
80//!     }
81//! }
82//!
83//! impl SampleUniform for MyF32 {
84//!     type Sampler = UniformMyF32;
85//! }
86//!
87//! let (low, high) = (MyF32(17.0f32), MyF32(22.0f32));
88//! let uniform = Uniform::new(low, high);
89//! let x = uniform.sample(&mut thread_rng());
90//! ```
91//!
92//! [`Uniform`]: struct.Uniform.html
93//! [`Rng::gen_range`]: ../../trait.Rng.html#method.gen_range
94//! [`SampleUniform`]: trait.SampleUniform.html
95//! [`UniformSampler`]: trait.UniformSampler.html
96//! [`UniformInt`]: struct.UniformInt.html
97//! [`UniformFloat`]: struct.UniformFloat.html
98//! [`UniformDuration`]: struct.UniformDuration.html

107/// Sample values uniformly between two bounds.
108///
109/// [`Uniform::new`] and [`Uniform::new_inclusive`] construct a uniform
110/// distribution sampling from the given range; these functions may do extra
111/// work up front to make sampling of multiple values faster.
112///
113/// When sampling from a constant range, many calculations can happen at
114/// compile-time and all methods should be fast; for floating-point ranges and
115/// the full range of integer types this should have comparable performance to
116/// the `Standard` distribution.
117///
118/// Steps are taken to avoid bias which might be present in naive
119/// implementations; for example `rng.gen::<u8>() % 170` samples from the range
120/// `[0, 169]` but is twice as likely to select numbers less than 85 than other
121/// values. Further, the implementations here give more weight to the high-bits
122/// generated by the RNG than the low bits, since with some RNGs the low-bits
123/// are of lower quality than the high bits.
124///
125/// Implementations should attempt to sample in `[low, high)` for
126/// `Uniform::new(low, high)`, i.e., excluding `high`, but this may be very
127/// difficult. All the primitive integer types satisfy this property, and the
128/// float types normally satisfy it, but rounding may mean `high` can occur.
129///
130/// # Example
131///
132/// ```
133/// use rand::distributions::{Distribution, Uniform};
134///
135/// fn main() {
136///     let between = Uniform::from(10..10000);
137///     let mut rng = rand::thread_rng();
138///     let mut sum = 0;
139///     for _ in 0..1000 {
140///         sum += between.sample(&mut rng);
141///     }
142///     println!("{}", sum);
143/// }
144/// ```
145///
146/// [`Uniform::new`]: struct.Uniform.html#method.new
147/// [`Uniform::new_inclusive`]: struct.Uniform.html#method.new_inclusive
148/// [`new`]: struct.Uniform.html#method.new
149/// [`new_inclusive`]: struct.Uniform.html#method.new_inclusive
150#[derive(Clone, Copy, Debug)]
151pub struct Uniform<X: SampleUniform> {
152    inner: X::Sampler,
153}
154
155impl<X: SampleUniform> Uniform<X> {
156    /// Create a new `Uniform` instance which samples uniformly from the half
157    /// open range `[low, high)` (excluding `high`). Panics if `low >= high`.
158    pub fn new(low: X, high: X) -> Uniform<X> {
159        Uniform { inner: X::Sampler::new(low, high) }
160    }
161
162    /// Create a new `Uniform` instance which samples uniformly from the closed
163    /// range `[low, high]` (inclusive). Panics if `low > high`.
164    pub fn new_inclusive(low: X, high: X) -> Uniform<X> {
165        Uniform { inner: X::Sampler::new_inclusive(low, high) }
166    }
167}
168
169impl<X: SampleUniform> Distribution<X> for Uniform<X> {
170    fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> X {
171        self.inner.sample(rng)
172    }
173}

175/// Helper trait for creating objects using the correct implementation of
176/// [`UniformSampler`] for the sampling type.
177///
178/// See the [module documentation] on how to implement [`Uniform`] range
179/// sampling for a custom type.
180///
181/// [`UniformSampler`]: trait.UniformSampler.html
182/// [module documentation]: index.html
183/// [`Uniform`]: struct.Uniform.html
184pub trait SampleUniform: Sized {
185    /// The `UniformSampler` implementation supporting type `X`.
186    type Sampler: UniformSampler<X = Self>;
187}
188
189/// Helper trait handling actual uniform sampling.
190///
191/// See the [module documentation] on how to implement [`Uniform`] range
192/// sampling for a custom type.
193///
194/// Implementation of [`sample_single`] is optional, and is only useful when
195/// the implementation can be faster than `Self::new(low, high).sample(rng)`.
196///
197/// [module documentation]: index.html
198/// [`Uniform`]: struct.Uniform.html
199/// [`sample_single`]: trait.UniformSampler.html#method.sample_single
200pub trait UniformSampler: Sized {
201    /// The type sampled by this implementation.
202    type X;
203
204    /// Construct self, with inclusive lower bound and exclusive upper bound
205    /// `[low, high)`.
206    ///
207    /// Usually users should not call this directly but instead use
208    /// `Uniform::new`, which asserts that `low < high` before calling this.
209    fn new(low: Self::X, high: Self::X) -> Self;
210
211    /// Construct self, with inclusive bounds `[low, high]`.
212    ///
213    /// Usually users should not call this directly but instead use
214    /// `Uniform::new_inclusive`, which asserts that `low <= high` before
215    /// calling this.
216    fn new_inclusive(low: Self::X, high: Self::X) -> Self;
217
218    /// Sample a value.
219    fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> Self::X;
220
221    /// Sample a single value uniformly from a range with inclusive lower bound
222    /// and exclusive upper bound `[low, high)`.
223    ///
224    /// Usually users should not call this directly but instead use
225    /// `Uniform::sample_single`, which asserts that `low < high` before calling
226    /// this.
227    ///
228    /// Via this method, implementations can provide a method optimized for
229    /// sampling only a single value from the specified range. The default
230    /// implementation simply calls `UniformSampler::new` then `sample` on the
231    /// result.
232    fn sample_single<R: Rng + ?Sized>(low: Self::X, high: Self::X, rng: &mut R)
233        -> Self::X
234    {
235        let uniform: Self = UniformSampler::new(low, high);
236        uniform.sample(rng)
237    }
238}

240impl<X: SampleUniform> From<::core::ops::Range<X>> for Uniform<X> {
241    fn from(r: ::core::ops::Range<X>) -> Uniform<X> {
242        Uniform::new(r.start, r.end)
243    }
244}

251/// The back-end implementing [`UniformSampler`] for integer types.
252///
253/// Unless you are implementing [`UniformSampler`] for your own type, this type
254/// should not be used directly, use [`Uniform`] instead.
255///
256/// # Implementation notes
257///
258/// For a closed range, the number of possible numbers we should generate is
259/// `range = (high - low + 1)`. It is not possible to end up with a uniform
260/// distribution if we map *all* the random integers that can be generated to
261/// this range. We have to map integers from a `zone` that is a multiple of the
262/// range. The rest of the integers, that cause a bias, are rejected.
263///
264/// The problem with `range` is that to cover the full range of the type, it has
265/// to store `unsigned_max + 1`, which can't be represented. But if the range
266/// covers the full range of the type, no modulus is needed. A range of size 0
267/// can't exist, so we use that to represent this special case. Wrapping
268/// arithmetic even makes representing `unsigned_max + 1` as 0 simple.
269///
270/// We don't calculate `zone` directly, but first calculate the number of
271/// integers to reject. To handle `unsigned_max + 1` not fitting in the type,
272/// we use:
273/// `ints_to_reject = (unsigned_max + 1) % range;`
274/// `ints_to_reject = (unsigned_max - range + 1) % range;`
275///
276/// The smallest integer PRNGs generate is `u32`. That is why for small integer
277/// sizes (`i8`/`u8` and `i16`/`u16`) there is an optimization: don't pick the
278/// largest zone that can fit in the small type, but pick the largest zone that
279/// can fit in an `u32`. `ints_to_reject` is always less than half the size of
280/// the small integer. This means the first bit of `zone` is always 1, and so
281/// are all the other preceding bits of a larger integer. The easiest way to
282/// grow the `zone` for the larger type is to simply sign extend it.
283///
284/// An alternative to using a modulus is widening multiply: After a widening
285/// multiply by `range`, the result is in the high word. Then comparing the low
286/// word against `zone` makes sure our distribution is uniform.
287///
288/// [`UniformSampler`]: trait.UniformSampler.html
289/// [`Uniform`]: struct.Uniform.html
290#[derive(Clone, Copy, Debug)]
291pub struct UniformInt<X> {
292    low: X,
293    range: X,
294    zone: X,
295}
296
297macro_rules! uniform_int_impl {
298    ($ty:ty, $signed:ty, $unsigned:ident,
299     $i_large:ident, $u_large:ident) => {
300        impl SampleUniform for $ty {
301            type Sampler = UniformInt<$ty>;
302        }
303
304        impl UniformSampler for UniformInt<$ty> {
305            // We play free and fast with unsigned vs signed here
306            // (when $ty is signed), but that's fine, since the
307            // contract of this macro is for $ty and $unsigned to be
308            // "bit-equal", so casting between them is a no-op.
309
310            type X = $ty;
311
312            #[inline] // if the range is constant, this helps LLVM to do the
313                      // calculations at compile-time.
314            fn new(low: Self::X, high: Self::X) -> Self {
315                assert!(low < high, "Uniform::new called with `low >= high`");
316                UniformSampler::new_inclusive(low, high - 1)
317            }
318
319            #[inline] // if the range is constant, this helps LLVM to do the
320                      // calculations at compile-time.
321            fn new_inclusive(low: Self::X, high: Self::X) -> Self {
322                assert!(low <= high,
323                        "Uniform::new_inclusive called with `low > high`");
324                let unsigned_max = ::core::$unsigned::MAX;
325
326                let range = high.wrapping_sub(low).wrapping_add(1) as $unsigned;
327                let ints_to_reject =
328                    if range > 0 {
329                        (unsigned_max - range + 1) % range
330                    } else {
331                        0
332                    };
333                let zone = unsigned_max - ints_to_reject;
334
335                UniformInt {
336                    low: low,
337                    // These are really $unsigned values, but store as $ty:
338                    range: range as $ty,
339                    zone: zone as $ty
340                }
341            }
342
343            fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> Self::X {
344                let range = self.range as $unsigned as $u_large;
345                if range > 0 {
346                    // Grow `zone` to fit a type of at least 32 bits, by
347                    // sign-extending it (the first bit is always 1, so are all
348                    // the preceding bits of the larger type).
349                    // For types that already have the right size, all the
350                    // casting is a no-op.
351                    let zone = self.zone as $signed as $i_large as $u_large;
352                    loop {
353                        let v: $u_large = rng.gen();
354                        let (hi, lo) = v.wmul(range);
355                        if lo <= zone {
356                            return self.low.wrapping_add(hi as $ty);
357                        }
358                    }
359                } else {
360                    // Sample from the entire integer range.
361                    rng.gen()
362                }
363            }
364
365            fn sample_single<R: Rng + ?Sized>(low: Self::X,
366                                              high: Self::X,
367                                              rng: &mut R) -> Self::X
368            {
369                assert!(low < high,
370                        "Uniform::sample_single called with low >= high");
371                let range = high.wrapping_sub(low) as $unsigned as $u_large;
372                let zone =
373                    if ::core::$unsigned::MAX <= ::core::u16::MAX as $unsigned {
374                        // Using a modulus is faster than the approximation for
375                        // i8 and i16. I suppose we trade the cost of one
376                        // modulus for near-perfect branch prediction.
377                        let unsigned_max: $u_large = ::core::$u_large::MAX;
378                        let ints_to_reject = (unsigned_max - range + 1) % range;
379                        unsigned_max - ints_to_reject
380                    } else {
381                        // conservative but fast approximation
382                       range << range.leading_zeros()
383                    };
384
385                loop {
386                    let v: $u_large = rng.gen();
387                    let (hi, lo) = v.wmul(range);
388                    if lo <= zone {
389                        return low.wrapping_add(hi as $ty);
390                    }
391                }
392            }
393        }
394    }
395}

413trait WideningMultiply<RHS = Self> {
414    type Output;
415
416    fn wmul(self, x: RHS) -> Self::Output;
417}
418
419macro_rules! wmul_impl {
420    ($ty:ty, $wide:ty, $shift:expr) => {
421        impl WideningMultiply for $ty {
422            type Output = ($ty, $ty);
423
424            #[inline(always)]
425            fn wmul(self, x: $ty) -> Self::Output {
426                let tmp = (self as $wide) * (x as $wide);
427                ((tmp >> $shift) as $ty, tmp as $ty)
428            }
429        }
430    }
431}
432wmul_impl! { u8, u16, 8 }
433wmul_impl! { u16, u32, 16 }
434wmul_impl! { u32, u64, 32 }
435#[cfg(feature = "i128_support")]
436wmul_impl! { u64, u128, 64 }

438// This code is a translation of the __mulddi3 function in LLVM's
439// compiler-rt. It is an optimised variant of the common method
440// `(a + b) * (c + d) = ac + ad + bc + bd`.
441//
442// For some reason LLVM can optimise the C version very well, but
443// keeps shuffeling registers in this Rust translation.
444macro_rules! wmul_impl_large {
445    ($ty:ty, $half:expr) => {
446        impl WideningMultiply for $ty {
447            type Output = ($ty, $ty);
448
449            #[inline(always)]
450            fn wmul(self, b: $ty) -> Self::Output {
451                const LOWER_MASK: $ty = !0 >> $half;
452                let mut low = (self & LOWER_MASK).wrapping_mul(b & LOWER_MASK);
453                let mut t = low >> $half;
454                low &= LOWER_MASK;
455                t += (self >> $half).wrapping_mul(b & LOWER_MASK);
456                low += (t & LOWER_MASK) << $half;
457                let mut high = t >> $half;
458                t = low >> $half;
459                low &= LOWER_MASK;
460                t += (b >> $half).wrapping_mul(self & LOWER_MASK);
461                low += (t & LOWER_MASK) << $half;
462                high += t >> $half;
463                high += (self >> $half).wrapping_mul(b >> $half);
464
465                (high, low)
466            }
467        }
468    }
469}
470#[cfg(not(feature = "i128_support"))]
471wmul_impl_large! { u64, 32 }
472#[cfg(feature = "i128_support")]
473wmul_impl_large! { u128, 64 }
474
475macro_rules! wmul_impl_usize {
476    ($ty:ty) => {
477        impl WideningMultiply for usize {

479
480            #[inline(always)]
481            fn wmul(self, x: usize) -> Self::Output {
482                let (high, low) = (self as $ty).wmul(x as $ty);
483                (high as usize, low as usize)
484            }
485        }
486    }

495/// The back-end implementing [`UniformSampler`] for floating-point types.
496///
497/// Unless you are implementing [`UniformSampler`] for your own type, this type
498/// should not be used directly, use [`Uniform`] instead.
499///
500/// # Implementation notes
501///
502/// Instead of generating a float in the `[0, 1)` range using [`Standard`], the
503/// `UniformFloat` implementation converts the output of an PRNG itself. This
504/// way one or two steps can be optimized out.
505///
506/// The floats are first converted to a value in the `[1, 2)` interval using a
507/// transmute-based method, and then mapped to the expected range with a
508/// multiply and addition. Values produced this way have what equals 22 bits of
509/// random digits for an `f32`, and 52 for an `f64`.
510///
511/// Currently there is no difference between [`new`] and [`new_inclusive`],
512/// because the boundaries of a floats range are a bit of a fuzzy concept due to
513/// rounding errors.
514///
515/// [`UniformSampler`]: trait.UniformSampler.html
516/// [`new`]: trait.UniformSampler.html#tymethod.new
517/// [`new_inclusive`]: trait.UniformSampler.html#tymethod.new_inclusive
518/// [`Uniform`]: struct.Uniform.html
519/// [`Standard`]: ../struct.Standard.html
520#[derive(Clone, Copy, Debug)]
521pub struct UniformFloat<X> {
522    scale: X,
523    offset: X,
524}
525
526macro_rules! uniform_float_impl {
527    ($ty:ty, $bits_to_discard:expr, $next_u:ident) => {
528        impl SampleUniform for $ty {
529            type Sampler = UniformFloat<$ty>;
530        }
531
532        impl UniformSampler for UniformFloat<$ty> {
533            type X = $ty;
534
535            fn new(low: Self::X, high: Self::X) -> Self {
536                assert!(low < high, "Uniform::new called with `low >= high`");
537                let scale = high - low;
538                let offset = low - scale;
539                UniformFloat {
540                    scale: scale,
541                    offset: offset,
542                }
543            }
544
545            fn new_inclusive(low: Self::X, high: Self::X) -> Self {
546                assert!(low <= high,
547                        "Uniform::new_inclusive called with `low > high`");
548                let scale = high - low;
549                let offset = low - scale;
550                UniformFloat {
551                    scale: scale,
552                    offset: offset,
553                }
554            }
555
556            fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> Self::X {
557                // Generate a value in the range [1, 2)
558                let value1_2 = (rng.$next_u() >> $bits_to_discard)
559                               .into_float_with_exponent(0);
560                // We don't use `f64::mul_add`, because it is not available with
561                // `no_std`. Furthermore, it is slower for some targets (but
562                // faster for others). However, the order of multiplication and
563                // addition is important, because on some platforms (e.g. ARM)
564                // it will be optimized to a single (non-FMA) instruction.
565                value1_2 * self.scale + self.offset
566            }
567
568            fn sample_single<R: Rng + ?Sized>(low: Self::X,
569                                              high: Self::X,
570                                              rng: &mut R) -> Self::X {
571                assert!(low < high,
572                        "Uniform::sample_single called with low >= high");
573                let scale = high - low;
574                let offset = low - scale;
575                // Generate a value in the range [1, 2)
576                let value1_2 = (rng.$next_u() >> $bits_to_discard)
577                               .into_float_with_exponent(0);
578                // Doing multiply before addition allows some architectures to
579                // use a single instruction.
580                value1_2 * scale + offset
581            }
582        }
583    }
584}

586uniform_float_impl! { f32, 32 - 23, next_u32 }
587uniform_float_impl! { f64, 64 - 52, next_u64 }

630    }

632    #[inline]
633    fn new_inclusive(low: Duration, high: Duration) -> UniformDuration {
634        assert!(low <= high, "Uniform::new_inclusive called with `low > high`");
635        let size = high - low;
636        let nanos = size
637            .as_secs()
638            .checked_mul(1_000_000_000)
639            .and_then(|n| n.checked_add(size.subsec_nanos() as u64));
640
641        let mode = match nanos {
642            Some(nanos) => {
643                UniformDurationMode::Small {
644                    nanos: Uniform::new_inclusive(0, nanos),
645                }
646            }
647            None => {
648                UniformDurationMode::Large {
649                    size: size,
650                    secs: Uniform::new_inclusive(0, size.as_secs()),
651                }
652            }
653        };
654
655        UniformDuration {
656            mode,
657            offset: low,
658        }
659    }
660
661    #[inline]
662    fn sample<R: Rng + ?Sized>(&self, rng: &mut R) -> Duration {
663        let d = match self.mode {
664            UniformDurationMode::Small { nanos } => {
665                let nanos = nanos.sample(rng);
666                Duration::new(nanos / 1_000_000_000, (nanos % 1_000_000_000) as u32)
667            }
668            UniformDurationMode::Large { size, secs } => {
669                // constant folding means this is at least as fast as `gen_range`
670                let nano_range = Uniform::new(0, 1_000_000_000);
671                loop {
672                    let d = Duration::new(secs.sample(rng), nano_range.sample(rng));
673                    if d <= size {
674                        break d;
675                    }
676                }
677            }
678        };
679
680        self.offset + d
681    }
