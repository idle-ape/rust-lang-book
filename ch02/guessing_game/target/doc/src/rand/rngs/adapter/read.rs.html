11//! A wrapper around any Read to treat it as an RNG.
12
13use std::io::Read;
14
15use rand_core::{RngCore, Error, ErrorKind, impls};
16
17
18/// An RNG that reads random bytes straight from any type supporting
19/// `std::io::Read`, for example files.
20///
21/// This will work best with an infinite reader, but that is not required.
22///
23/// This can be used with `/dev/urandom` on Unix but it is recommended to use
24/// [`OsRng`] instead.
25///
26/// # Panics
27///
28/// `ReadRng` uses `std::io::read_exact`, which retries on interrupts. All other
29/// errors from the underlying reader, including when it does not have enough
30/// data, will only be reported through [`try_fill_bytes`]. The other
31/// [`RngCore`] methods will panic in case of an error.
32///
33/// # Example
34///
35/// ```
36/// use rand::{read, Rng};
37///
38/// let data = vec![1, 2, 3, 4, 5, 6, 7, 8];
39/// let mut rng = read::ReadRng::new(&data[..]);
40/// println!("{:x}", rng.gen::<u32>());
41/// ```
42///
43/// [`OsRng`]: ../struct.OsRng.html
44/// [`RngCore`]: ../../trait.RngCore.html
45/// [`try_fill_bytes`]: ../../trait.RngCore.html#method.tymethod.try_fill_bytes
46#[derive(Debug)]
47pub struct ReadRng<R> {
48    reader: R
49}
50
51impl<R: Read> ReadRng<R> {
52    /// Create a new `ReadRng` from a `Read`.
53    pub fn new(r: R) -> ReadRng<R> {
54        ReadRng {
55            reader: r
56        }
57    }
58}
59
60impl<R: Read> RngCore for ReadRng<R> {
61    fn next_u32(&mut self) -> u32 {
62        impls::next_u32_via_fill(self)
63    }
64
65    fn next_u64(&mut self) -> u64 {
66        impls::next_u64_via_fill(self)
67    }
68
69    fn fill_bytes(&mut self, dest: &mut [u8]) {
70        self.try_fill_bytes(dest).unwrap_or_else(|err|
71                panic!("reading random bytes from Read implementation failed; error: {}", err));
72    }
73
74    fn try_fill_bytes(&mut self, dest: &mut [u8]) -> Result<(), Error> {
75        if dest.len() == 0 { return Ok(()); }
76        // Use `std::io::read_exact`, which retries on `ErrorKind::Interrupted`.
77        self.reader.read_exact(dest).map_err(|err| {
78            match err.kind() {
79                ::std::io::ErrorKind::UnexpectedEof => Error::with_cause(
80                    ErrorKind::Unavailable,
81                    "not enough bytes available, reached end of source", err),
82                _ => Error::with_cause(ErrorKind::Unavailable,
83                    "error reading from Read source", err)
84            }
85        })
86    }
87}

89#[cfg(test)]
90mod test {
91    use super::ReadRng;
92    use {RngCore, ErrorKind};
93
94    #[test]
95    fn test_reader_rng_u64() {
96        // transmute from the target to avoid endianness concerns.
97        let v = vec![0u8, 0, 0, 0, 0, 0, 0, 1,
98                     0  , 0, 0, 0, 0, 0, 0, 2,
99                     0,   0, 0, 0, 0, 0, 0, 3];
100        let mut rng = ReadRng::new(&v[..]);
101
102        assert_eq!(rng.next_u64(), 1_u64.to_be());
103        assert_eq!(rng.next_u64(), 2_u64.to_be());
104        assert_eq!(rng.next_u64(), 3_u64.to_be());
105    }
106
107    #[test]
108    fn test_reader_rng_u32() {
109        let v = vec![0u8, 0, 0, 1, 0, 0, 0, 2, 0, 0, 0, 3];
110        let mut rng = ReadRng::new(&v[..]);
111
112        assert_eq!(rng.next_u32(), 1_u32.to_be());
113        assert_eq!(rng.next_u32(), 2_u32.to_be());
114        assert_eq!(rng.next_u32(), 3_u32.to_be());
115    }
116
117    #[test]
118    fn test_reader_rng_fill_bytes() {
119        let v = [1u8, 2, 3, 4, 5, 6, 7, 8];
120        let mut w = [0u8; 8];
121
122        let mut rng = ReadRng::new(&v[..]);
123        rng.fill_bytes(&mut w);
124
125        assert!(v == w);
126    }
127
128    #[test]
129    fn test_reader_rng_insufficient_bytes() {
130        let v = [1u8, 2, 3, 4, 5, 6, 7, 8];
131        let mut w = [0u8; 9];
132
133        let mut rng = ReadRng::new(&v[..]);
134
135        assert!(rng.try_fill_bytes(&mut w).err().unwrap().kind == ErrorKind::Unavailable);
136    }
137}
