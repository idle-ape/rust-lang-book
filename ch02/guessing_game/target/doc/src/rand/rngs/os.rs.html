11//! Interface to the random number generator of the operating system.
12
13use std::fmt;
14use rand_core::{CryptoRng, RngCore, Error, impls};

16/// A random number generator that retrieves randomness straight from the
17/// operating system.
18///
19/// This is the preferred external source of entropy for most applications.
20/// Commonly it is used to initialize a user-space RNG, which can then be used
21/// to generate random values with much less overhead than `OsRng`.
22///
23/// You may prefer to use [`EntropyRng`] instead of `OsRng`. It is unlikely, but
24/// not entirely theoretical, for `OsRng` to fail. In such cases [`EntropyRng`]
25/// falls back on a good alternative entropy source.
26///
27/// `OsRng::new()` is guaranteed to be very cheap (after the first successful
28/// call), and will never consume more than one file handle per process.
29///
30/// # Platform sources
31///
32/// | OS               | interface
33/// |------------------|---------------------------------------------------------
34/// | Linux, Android   | [`getrandom`][1] system call if available, otherwise [`/dev/urandom`][2] after reading from `/dev/random` once
35/// | Windows          | [`RtlGenRandom`][3]
36/// | macOS, iOS       | [`SecRandomCopyBytes`][4]
37/// | FreeBSD          | [`kern.arandom`][5]
38/// | OpenBSD, Bitrig  | [`getentropy`][6]
39/// | NetBSD           | [`/dev/urandom`][7] after reading from `/dev/random` once
40/// | Dragonfly BSD    | [`/dev/random`][8]
41/// | Solaris, illumos | [`getrandom`][9] system call if available, otherwise [`/dev/random`][10]
42/// | Fuchsia OS       | [`cprng_draw`][11]
43/// | Redox            | [`rand:`][12]
44/// | CloudABI         | [`random_get`][13]
45/// | Haiku            | `/dev/random` (identical to `/dev/urandom`)
46/// | Web browsers     | [`Crypto.getRandomValues`][14] (see [Support for WebAssembly and ams.js][14])
47/// | Node.js          | [`crypto.randomBytes`][15] (see [Support for WebAssembly and ams.js][16])
48///
49/// Rand doesn't have a blanket implementation for all Unix-like operating
50/// systems that reads from `/dev/urandom`. This ensures all supported operating
51/// systems are using the recommended interface and respect maximum buffer
52/// sizes.
53///
54/// ## Support for WebAssembly and ams.js
55///
56/// The three Emscripten targets `asmjs-unknown-emscripten`,
57/// `wasm32-unknown-emscripten` and `wasm32-experimental-emscripten` use
58/// Emscripten's emulation of `/dev/random` on web browsers and Node.js.
59/// Unfortunately it falls back to the insecure `Math.random()` if a browser
60/// doesn't support [`Crypto.getRandomValues`][12].
61///
62/// The bare Wasm target `wasm32-unknown-unknown` tries to call the javascript
63/// methods directly, using `stdweb` in combination with `cargo-web`.
64/// `wasm-bindgen` is not yet supported.
65///
66/// ## Early boot
67///
68/// It is possible that early in the boot process the OS hasn't had enough time
69/// yet to collect entropy to securely seed its RNG, especially on virtual
70/// machines.
71///
72/// Some operating systems always block the thread until the RNG is securely
73/// seeded. This can take anywhere from a few seconds to more than a minute.
74/// Others make a best effort to use a seed from before the shutdown and don't
75/// document much.
76///
77/// A few, Linux, NetBSD and Solaris, offer a choice between blocking, and
78/// getting an error. With `try_fill_bytes` we choose to get the error
79/// ([`ErrorKind::NotReady`]), while the other methods use a blocking interface.
80///
81/// On Linux (when the `genrandom` system call is not available) and on NetBSD
82/// reading from `/dev/urandom` never blocks, even when the OS hasn't collected
83/// enough entropy yet. As a countermeasure we try to do a single read from
84/// `/dev/random` until we know the OS RNG is initialized (and store this in a
85/// global static).
86///
87/// # Panics
88///
89/// `OsRng` is extremely unlikely to fail if `OsRng::new()`, and one read from
90/// it, where succesfull. But in case it does fail, only [`try_fill_bytes`] is
91/// able to report the cause. Depending on the error the other [`RngCore`]
92/// methods will retry several times, and panic in case the error remains.
93///
94/// [`EntropyRng`]: struct.EntropyRng.html
95/// [`RngCore`]: ../trait.RngCore.html
96/// [`try_fill_bytes`]: ../trait.RngCore.html#method.tymethod.try_fill_bytes
97/// [`ErrorKind::NotReady`]: ../enum.ErrorKind.html#variant.NotReady
98///
99/// [1]: http://man7.org/linux/man-pages/man2/getrandom.2.html
100/// [2]: http://man7.org/linux/man-pages/man4/urandom.4.html
101/// [3]: https://msdn.microsoft.com/en-us/library/windows/desktop/aa387694.aspx
102/// [4]: https://developer.apple.com/documentation/security/1399291-secrandomcopybytes?language=objc
103/// [5]: https://www.freebsd.org/cgi/man.cgi?query=random&sektion=4
104/// [6]: https://man.openbsd.org/getentropy.2
105/// [7]: http://netbsd.gw.com/cgi-bin/man-cgi?random+4+NetBSD-current
106/// [8]: https://leaf.dragonflybsd.org/cgi/web-man?command=random&section=4
107/// [9]: https://docs.oracle.com/cd/E88353_01/html/E37841/getrandom-2.html
108/// [10]: https://docs.oracle.com/cd/E86824_01/html/E54777/random-7d.html
109/// [11]: https://fuchsia.googlesource.com/zircon/+/HEAD/docs/syscalls/cprng_draw.md
110/// [12]: https://github.com/redox-os/randd/blob/master/src/main.rs
111/// [13]: https://github.com/NuxiNL/cloudabi/blob/v0.20/cloudabi.txt#L1826
112/// [14]: https://www.w3.org/TR/WebCryptoAPI/#Crypto-method-getRandomValues

135impl RngCore for OsRng {
136    fn next_u32(&mut self) -> u32 {
137        impls::next_u32_via_fill(self)
138    }
139
140    fn next_u64(&mut self) -> u64 {
141        impls::next_u64_via_fill(self)
142    }
143
144    fn fill_bytes(&mut self, dest: &mut [u8]) {
145        use std::{time, thread};
146
147        // We cannot return Err(..), so we try to handle before panicking.
148        const MAX_RETRY_PERIOD: u32 = 10; // max 10s
149        const WAIT_DUR_MS: u32 = 100; // retry every 100ms
150        let wait_dur = time::Duration::from_millis(WAIT_DUR_MS as u64);
151        const RETRY_LIMIT: u32 = (MAX_RETRY_PERIOD * 1000) / WAIT_DUR_MS;
152        const TRANSIENT_RETRIES: u32 = 8;
153        let mut err_count = 0;
154        let mut error_logged = false;
155
156        // Maybe block until the OS RNG is initialized
157        let mut read = 0;
158        if let Ok(n) = self.0.test_initialized(dest, true) { read = n };
159        let dest = &mut dest[read..];
160
161        loop {
162            if let Err(e) = self.try_fill_bytes(dest) {
163                if err_count >= RETRY_LIMIT {
164                    error!("OsRng failed too many times; last error: {}", e);
165                    panic!("OsRng failed too many times; last error: {}", e);
166                }
167
168                if e.kind.should_wait() {
169                    if !error_logged {
170                        warn!("OsRng failed; waiting up to {}s and retrying. Error: {}",
171                                MAX_RETRY_PERIOD, e);
172                        error_logged = true;
173                    }
174                    err_count += 1;
175                    thread::sleep(wait_dur);
176                    continue;
177                } else if e.kind.should_retry() {
178                    if !error_logged {
179                        warn!("OsRng failed; retrying up to {} times. Error: {}",
180                                TRANSIENT_RETRIES, e);
181                        error_logged = true;
182                    }
183                    err_count += (RETRY_LIMIT + TRANSIENT_RETRIES - 1)
184                            / TRANSIENT_RETRIES;    // round up
185                    continue;
186                } else {
187                    error!("OsRng failed: {}", e);
188                    panic!("OsRng fatal error: {}", e);
189                }
190            }
191
192            break;
193        }
194    }
195
196    fn try_fill_bytes(&mut self, dest: &mut [u8]) -> Result<(), Error> {
197        // Some systems do not support reading 0 random bytes.
198        // (And why waste a system call?)
199        if dest.len() == 0 { return Ok(()); }
200
201        let read = self.0.test_initialized(dest, false)?;
202        let dest = &mut dest[read..];
203
204        let max = self.0.max_chunk_size();
205        if dest.len() <= max {
206            trace!("OsRng: reading {} bytes via {}",
207                   dest.len(), self.0.method_str());
208        } else {
209            trace!("OsRng: reading {} bytes via {} in {} chunks of {} bytes",
210                   dest.len(), self.0.method_str(), (dest.len() + max) / max, max);
211        }
212        for slice in dest.chunks_mut(max) {
213            self.0.fill_chunk(slice)?;
214        }
215        Ok(())
216    }
217}

219trait OsRngImpl where Self: Sized {
220    // Create a new `OsRng` platform interface.
221    fn new() -> Result<Self, Error>;
222
223    // Fill a chunk with random bytes.
224    fn fill_chunk(&mut self, dest: &mut [u8]) -> Result<(), Error>;
225
226    // Test whether the OS RNG is initialized. This method may not be possible
227    // to support cheaply (or at all) on all operating systems.
228    //
229    // If `blocking` is set, this will cause the OS the block execution until
230    // its RNG is initialized.
231    //
232    // Random values that are read while this are stored in `dest`, the amount
233    // of read bytes is returned.
234    fn test_initialized(&mut self, _dest: &mut [u8], _blocking: bool)
235        -> Result<usize, Error> { Ok(0) }
236
237    // Maximum chunk size supported.
238    fn max_chunk_size(&self) -> usize { ::core::usize::MAX }
239
240    // Name of the OS interface (used for logging).
241    fn method_str(&self) -> &'static str;
242}

247// Helper functions to read from a random device such as `/dev/urandom`.
248//
249// All instances use a single internal file handle, to prevent possible
250// exhaustion of file descriptors.
251#[cfg(any(target_os = "linux", target_os = "android",
252          target_os = "netbsd", target_os = "dragonfly",
253          target_os = "solaris", target_os = "redox",
254          target_os = "haiku", target_os = "emscripten"))]
255mod random_device {
256    use {Error, ErrorKind};
257    use std::fs::File;
258    use std::io;
259    use std::io::Read;
260    use std::sync::{Once, Mutex, ONCE_INIT};
261
262    // TODO: remove outer Option when `Mutex::new(None)` is a constant expression
263    static mut READ_RNG_FILE: Option<Mutex<Option<File>>> = None;
264    static READ_RNG_ONCE: Once = ONCE_INIT;
265
266    #[allow(unused)]
267    pub fn open<F>(path: &'static str, open_fn: F) -> Result<(), Error>
268        where F: Fn(&'static str) -> Result<File, io::Error>
269    {
270        READ_RNG_ONCE.call_once(|| {
271            unsafe { READ_RNG_FILE = Some(Mutex::new(None)) }
272        });
273
274        // We try opening the file outside the `call_once` fn because we cannot
275        // clone the error, thus we must retry on failure.
276
277        let mutex = unsafe { READ_RNG_FILE.as_ref().unwrap() };
278        let mut guard = mutex.lock().unwrap();
279        if (*guard).is_none() {
280            info!("OsRng: opening random device {}", path);
281            let file = open_fn(path).map_err(map_err)?;
282            *guard = Some(file);
283        };
284        Ok(())
285    }
286
287    pub fn read(dest: &mut [u8]) -> Result<(), Error> {
288        // We expect this function only to be used after `random_device::open`
289        // was succesful. Therefore we can assume that our memory was set with a
290        // valid object.
291        let mutex = unsafe { READ_RNG_FILE.as_ref().unwrap() };
292        let mut guard = mutex.lock().unwrap();
293        let file = (*guard).as_mut().unwrap();
294
295        // Use `std::io::read_exact`, which retries on `ErrorKind::Interrupted`.
296        file.read_exact(dest).map_err(|err| {
297            Error::with_cause(ErrorKind::Unavailable,
298                              "error reading random device", err)
299        })
300
301    }
302
303    pub fn map_err(err: io::Error) -> Error {
304        match err.kind() {
305            io::ErrorKind::Interrupted =>
306                    Error::new(ErrorKind::Transient, "interrupted"),
307            io::ErrorKind::WouldBlock =>
308                    Error::with_cause(ErrorKind::NotReady,
309                    "OS RNG not yet seeded", err),
310            _ => Error::with_cause(ErrorKind::Unavailable,
311                    "error while opening random device", err)
312        }
313    }
314}

319    extern crate libc;
320
321    use {Error, ErrorKind};
322    use super::random_device;

329    use std::sync::atomic::{AtomicBool, ATOMIC_BOOL_INIT, Ordering};

331
332    #[derive(Clone, Debug)]
333    pub struct OsRng {
334        method: OsRngMethod,
335        initialized: bool,
336    }
337
338    #[derive(Clone, Debug)]
339    enum OsRngMethod {
340        GetRandom,
341        RandomDevice,
342    }
343
344    impl OsRngImpl for OsRng {
345        fn new() -> Result<OsRng, Error> {
346            if is_getrandom_available() {
347                return Ok(OsRng { method: OsRngMethod::GetRandom,
348                                  initialized: false });
349            }
350            random_device::open("/dev/urandom", &|p| File::open(p))?;
351            Ok(OsRng { method: OsRngMethod::RandomDevice, initialized: false })
352        }
353
354        fn fill_chunk(&mut self, dest: &mut [u8]) -> Result<(), Error> {
355            match self.method {
356                OsRngMethod::GetRandom => getrandom_try_fill(dest, false),
357                OsRngMethod::RandomDevice => random_device::read(dest),
358            }
359        }
360
361        fn test_initialized(&mut self, dest: &mut [u8], blocking: bool)
362            -> Result<usize, Error>
363        {
364            static OS_RNG_INITIALIZED: AtomicBool = ATOMIC_BOOL_INIT;
365            if !self.initialized {
366                self.initialized = OS_RNG_INITIALIZED.load(Ordering::Relaxed);
367            }
368            if self.initialized { return Ok(0); }
369
370            let result = match self.method {
371                OsRngMethod::GetRandom => {
372                    getrandom_try_fill(dest, blocking)?;
373                    Ok(dest.len())
374                }
375                OsRngMethod::RandomDevice => {
376                    info!("OsRng: testing random device /dev/random");
377                    let mut file = OpenOptions::new()
378                        .read(true)
379                        .custom_flags(if blocking { 0 } else { libc::O_NONBLOCK })
380                        .open("/dev/random")
381                        .map_err(random_device::map_err)?;
382                    file.read(&mut dest[..1]).map_err(random_device::map_err)?;
383                    Ok(1)
384                }
385            };
386            OS_RNG_INITIALIZED.store(true, Ordering::Relaxed);
387            self.initialized = true;
388            result
389        }
390
391        fn method_str(&self) -> &'static str {
392            match self.method {
393                OsRngMethod::GetRandom => "getrandom",
394                OsRngMethod::RandomDevice => "/dev/urandom",
395            }
396        }
397    }

399    #[cfg(target_arch = "x86_64")]
400    const NR_GETRANDOM: libc::c_long = 318;
401    #[cfg(target_arch = "x86")]
402    const NR_GETRANDOM: libc::c_long = 355;
403    #[cfg(target_arch = "arm")]
404    const NR_GETRANDOM: libc::c_long = 384;
405    #[cfg(target_arch = "aarch64")]
406    const NR_GETRANDOM: libc::c_long = 278;
407     #[cfg(target_arch = "s390x")]
408    const NR_GETRANDOM: libc::c_long = 349;
409    #[cfg(target_arch = "powerpc")]
410    const NR_GETRANDOM: libc::c_long = 359;
411    #[cfg(target_arch = "mips")] // old ABI
412    const NR_GETRANDOM: libc::c_long = 4353;
413    #[cfg(target_arch = "mips64")]
414    const NR_GETRANDOM: libc::c_long = 5313;
415    #[cfg(not(any(target_arch = "x86_64", target_arch = "x86",
416                  target_arch = "arm", target_arch = "aarch64",
417                  target_arch = "s390x", target_arch = "powerpc",
418                  target_arch = "mips", target_arch = "mips64")))]
419    const NR_GETRANDOM: libc::c_long = 0;
420
421    fn getrandom(buf: &mut [u8], blocking: bool) -> libc::c_long {
422        extern "C" {
423            fn syscall(number: libc::c_long, ...) -> libc::c_long;
424        }
425        const GRND_NONBLOCK: libc::c_uint = 0x0001;
426
427        if NR_GETRANDOM == 0 { return -1 };
428
429        unsafe {
430            syscall(NR_GETRANDOM, buf.as_mut_ptr(), buf.len(),
431                    if blocking { 0 } else { GRND_NONBLOCK })
432        }
433    }
434
435    fn getrandom_try_fill(dest: &mut [u8], blocking: bool) -> Result<(), Error> {
436        let mut read = 0;
437        while read < dest.len() {
438            let result = getrandom(&mut dest[read..], blocking);
439            if result == -1 {
440                let err = io::Error::last_os_error();
441                let kind = err.kind();
442                if kind == io::ErrorKind::Interrupted {
443                    continue;
444                } else if kind == io::ErrorKind::WouldBlock {
445                    return Err(Error::with_cause(
446                        ErrorKind::NotReady,
447                        "getrandom not ready",
448                        err,
449                    ));
450                } else {
451                    return Err(Error::with_cause(
452                        ErrorKind::Unavailable,
453                        "unexpected getrandom error",
454                        err,
455                    ));
456                }
457            } else {
458                read += result as usize;
459            }
460        }
461        Ok(())
462    }
463
464    fn is_getrandom_available() -> bool {
465        static CHECKER: Once = ONCE_INIT;
466        static AVAILABLE: AtomicBool = ATOMIC_BOOL_INIT;
467
468        if NR_GETRANDOM == 0 { return false };
469
470        CHECKER.call_once(|| {
471            debug!("OsRng: testing getrandom");
472            let mut buf: [u8; 0] = [];
473            let result = getrandom(&mut buf, false);
474            let available = if result == -1 {
475                let err = io::Error::last_os_error().raw_os_error();
476                err != Some(libc::ENOSYS)
477            } else {
478                true
479            };
480            AVAILABLE.store(available, Ordering::Relaxed);
481            info!("OsRng: using {}", if available { "getrandom" } else { "/dev/urandom" });
482        });
483
484        AVAILABLE.load(Ordering::Relaxed)
485    }
486}

574// Read from `/dev/random`, with chunks of limited size (1040 bytes).
575// `/dev/random` uses the Hash_DRBG with SHA512 algorithm from NIST SP 800-90A.
576// `/dev/urandom` uses the FIPS 186-2 algorithm, which is considered less
577// secure. We choose to read from `/dev/random`.
578//
579// Since Solaris 11.3 the `getrandom` syscall is available. To make sure we can
580// compile on both Solaris and on OpenSolaris derivatives, that do not have the
581// function, we do a direct syscall instead of calling a library function.
582//
583// We have no way to differentiate between Solaris, illumos, SmartOS, etc.
584#[cfg(target_os = "solaris")]
585mod imp {
586    extern crate libc;
587
588    use {Error, ErrorKind};
589    use super::random_device;
590    use super::OsRngImpl;
591
592    use std::io;
593    use std::io::Read;
594    use std::fs::{File, OpenOptions};
595    use std::os::unix::fs::OpenOptionsExt;
596    use std::sync::atomic::{AtomicBool, ATOMIC_BOOL_INIT, Ordering};
597
598    #[derive(Clone, Debug)]
599    pub struct OsRng {
600        method: OsRngMethod,
601        initialized: bool,
602    }
603
604    #[derive(Clone, Debug)]
605    enum OsRngMethod {
606        GetRandom,
607        RandomDevice,
608    }
609
610    impl OsRngImpl for OsRng {
611        fn new() -> Result<OsRng, Error> {
612            if is_getrandom_available() {
613                return Ok(OsRng { method: OsRngMethod::GetRandom,
614                                  initialized: false });
615            }
616            let open = |p| OpenOptions::new()
617                .read(true)
618                .custom_flags(libc::O_NONBLOCK)
619                .open(p);
620            random_device::open("/dev/random", &open)?;
621            Ok(OsRng { method: OsRngMethod::RandomDevice, initialized: false })
622        }
623
624        fn fill_chunk(&mut self, dest: &mut [u8]) -> Result<(), Error> {
625            match self.method {
626                OsRngMethod::GetRandom => getrandom_try_fill(dest, false),
627                OsRngMethod::RandomDevice => random_device::read(dest),
628            }
629        }
630
631        fn test_initialized(&mut self, dest: &mut [u8], blocking: bool)
632            -> Result<usize, Error>
633        {
634            static OS_RNG_INITIALIZED: AtomicBool = ATOMIC_BOOL_INIT;
635            if !self.initialized {
636                self.initialized = OS_RNG_INITIALIZED.load(Ordering::Relaxed);
637            }
638            if self.initialized { return Ok(0); }
639
640            let chunk_len = ::core::cmp::min(1024, dest.len());
641            let dest = &mut dest[..chunk_len];
642
643            match self.method {
644                OsRngMethod::GetRandom => getrandom_try_fill(dest, blocking)?,
645                OsRngMethod::RandomDevice => {
646                    if blocking {
647                        info!("OsRng: testing random device /dev/random");
648                        // We already have a non-blocking handle, but now need a
649                        // blocking one. Not much choice except opening it twice
650                        let mut file = File::open("/dev/random")
651                            .map_err(random_device::map_err)?;
652                        file.read(dest).map_err(random_device::map_err)?;
653                    } else {
654                        self.fill_chunk(dest)?;
655                    }
656                }
657            };
658            OS_RNG_INITIALIZED.store(true, Ordering::Relaxed);
659            self.initialized = true;
660            Ok(chunk_len)
661        }
662
663        fn max_chunk_size(&self) -> usize {
664            // The documentation says 1024 is the maximum for getrandom, but
665            // 1040 for /dev/random.
666            1024
667        }
668
669        fn method_str(&self) -> &'static str {
670            match self.method {
671                OsRngMethod::GetRandom => "getrandom",
672                OsRngMethod::RandomDevice => "/dev/random",
673            }
674        }
675    }
676
677    fn getrandom(buf: &mut [u8], blocking: bool) -> libc::c_long {
678        extern "C" {
679            fn syscall(number: libc::c_long, ...) -> libc::c_long;
680        }
681
682        const SYS_GETRANDOM: libc::c_long = 143;
683        const GRND_NONBLOCK: libc::c_uint = 0x0001;
684        const GRND_RANDOM: libc::c_uint = 0x0002;
685
686        unsafe {
687            syscall(SYS_GETRANDOM, buf.as_mut_ptr(), buf.len(),
688                    if blocking { 0 } else { GRND_NONBLOCK } | GRND_RANDOM)
689        }
690    }
691
692    fn getrandom_try_fill(dest: &mut [u8], blocking: bool) -> Result<(), Error> {
693        let result = getrandom(dest, blocking);
694        if result == -1 || result == 0 {
695            let err = io::Error::last_os_error();
696            let kind = err.kind();
697            if kind == io::ErrorKind::WouldBlock {
698                return Err(Error::with_cause(
699                    ErrorKind::NotReady,
700                    "getrandom not ready",
701                    err,
702                ));
703            } else {
704                return Err(Error::with_cause(
705                    ErrorKind::Unavailable,
706                    "unexpected getrandom error",
707                    err,
708                ));
709            }
710        } else if result != dest.len() as i64 {
711            return Err(Error::new(ErrorKind::Unavailable,
712                                  "unexpected getrandom error"));
713        }
714        Ok(())
715    }
716
717    fn is_getrandom_available() -> bool {
718        use std::sync::atomic::{AtomicBool, ATOMIC_BOOL_INIT, Ordering};
719        use std::sync::{Once, ONCE_INIT};
720
721        static CHECKER: Once = ONCE_INIT;
722        static AVAILABLE: AtomicBool = ATOMIC_BOOL_INIT;
723
724        CHECKER.call_once(|| {
725            debug!("OsRng: testing getrandom");
726            let mut buf: [u8; 0] = [];
727            let result = getrandom(&mut buf, false);
728            let available = if result == -1 {
729                let err = io::Error::last_os_error().raw_os_error();
730                err != Some(libc::ENOSYS)
731            } else {
732                true
733            };
734            AVAILABLE.store(available, Ordering::Relaxed);
735            info!("OsRng: using {}", if available { "getrandom" } else { "/dev/random" });
736        });
737
738        AVAILABLE.load(Ordering::Relaxed)
739    }
740}

987#[cfg(all(target_arch = "wasm32",
988          not(target_os = "emscripten"),
989          feature = "stdweb"))]
990mod imp {
991    use std::mem;
992    use stdweb::unstable::TryInto;
993    use stdweb::web::error::Error as WebError;
994    use {Error, ErrorKind};
995    use super::OsRngImpl;
996
997    #[derive(Clone, Debug)]
998    enum OsRngMethod {
999        Browser,
1000        Node
1001    }
1002
1003    #[derive(Clone, Debug)]
1004    pub struct OsRng(OsRngMethod);
1005
1006    impl OsRngImpl for OsRng {
1007        fn new() -> Result<OsRng, Error> {
1008            let result = js! {
1009                try {
1010                    if (
1011                        typeof self === "object" &&
1012                        typeof self.crypto === "object" &&
1013                        typeof self.crypto.getRandomValues === "function"
1014                    ) {
1015                        return { success: true, ty: 1 };
1016                    }
1017
1018                    if (typeof require("crypto").randomBytes === "function") {
1019                        return { success: true, ty: 2 };
1020                    }
1021
1022                    return { success: false, error: new Error("not supported") };
1023                } catch(err) {
1024                    return { success: false, error: err };
1025                }
1026            };
1027
1028            if js!{ return @{ result.as_ref() }.success } == true {
1029                let ty = js!{ return @{ result }.ty };
1030
1031                if ty == 1 { Ok(OsRng(OsRngMethod::Browser)) }
1032                else if ty == 2 { Ok(OsRng(OsRngMethod::Node)) }
1033                else { unreachable!() }
1034            } else {
1035                let err: WebError = js!{ return @{ result }.error }.try_into().unwrap();
1036                Err(Error::with_cause(ErrorKind::Unavailable, "WASM Error", err))
1037            }
1038        }
1039
1040
1041        fn fill_chunk(&mut self, dest: &mut [u8]) -> Result<(), Error> {
1042            assert_eq!(mem::size_of::<usize>(), 4);
1043
1044            let len = dest.len() as u32;
1045            let ptr = dest.as_mut_ptr() as i32;
1046
1047            let result = match self.0 {
1048                OsRngMethod::Browser => js! {
1049                    try {
1050                        let array = new Uint8Array(@{ len });
1051                        self.crypto.getRandomValues(array);
1052                        HEAPU8.set(array, @{ ptr });
1053
1054                        return { success: true };
1055                    } catch(err) {
1056                        return { success: false, error: err };
1057                    }
1058                },
1059                OsRngMethod::Node => js! {
1060                    try {
1061                        let bytes = require("crypto").randomBytes(@{ len });
1062                        HEAPU8.set(new Uint8Array(bytes), @{ ptr });
1063
1064                        return { success: true };
1065                    } catch(err) {
1066                        return { success: false, error: err };
1067                    }
1068                }
1069            };
1070
1071            if js!{ return @{ result.as_ref() }.success } == true {
1072                Ok(())
1073            } else {
1074                let err: WebError = js!{ return @{ result }.error }.try_into().unwrap();
1075                Err(Error::with_cause(ErrorKind::Unexpected, "WASM Error", err))
1076            }
1077        }
1078
1079        fn max_chunk_size(&self) -> usize { 65536 }
1080
1081        fn method_str(&self) -> &'static str {
1082            match self.0 {
1083                OsRngMethod::Browser => "Crypto.getRandomValues",
1084                OsRngMethod::Node => "crypto.randomBytes",
1085            }
1086        }
1087    }
